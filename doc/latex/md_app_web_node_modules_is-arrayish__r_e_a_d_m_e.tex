\begin{quote}
Determines if an object can be used like an Array \end{quote}


\#\# Example 
\begin{DoxyCode}
var isArrayish = require('is-arrayish');

isArrayish([]); // true
isArrayish(\{\_\_proto\_\_: []\}); // true
isArrayish(\{\}); // false
isArrayish(\{length:10\}); // false
\end{DoxyCode}
