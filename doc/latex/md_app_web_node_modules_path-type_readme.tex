\begin{quote}
Check if a path is a file, directory, or symlink \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save path-type
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var pathType = require('path-type');

pathType.file('package.json').then(function (isFile) \{
    console.log(isFile);
    //=> true
\})
\end{DoxyCode}
