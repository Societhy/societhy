Access to v8\textquotesingle{}s \char`\"{}raw\char`\"{} {\ttfamily Call\+Site}s.

\subsection*{Installation}

\begin{DoxyVerb}$ npm install callsite
\end{DoxyVerb}


\subsection*{Example}


\begin{DoxyCode}
var stack = require('callsite');

foo();

function foo() \{
  bar();
\}

function bar() \{
  baz();
\}

function baz() \{
  console.log();
  stack().forEach(function(site)\{
    console.log('  \(\backslash\)033[36m%s\(\backslash\)033[90m in %s:%d\(\backslash\)033[0m'
      , site.getFunctionName() || 'anonymous'
      , site.getFileName()
      , site.getLineNumber());
  \});
  console.log();
\}
\end{DoxyCode}
