\begin{quote}
Easier stdin \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save get-stdin
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
// example.js
var stdin = require('get-stdin');

stdin(function (data) \{
    console.log(data);
    //=> unicorns
\});
\end{DoxyCode}



\begin{DoxyCode}
$ echo unicorns | node example.js
unicorns
\end{DoxyCode}
