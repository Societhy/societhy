Use this instead of `require(\textquotesingle{}os\textquotesingle{}).tmpdir()` to get a consistent behavior on different Node.\+js versions (even 0.\+8).

\subsection*{Install}


\begin{DoxyCode}
$ npm install --save os-tmpdir
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const osTmpdir = require('os-tmpdir');

osTmpdir();
//=> '/var/folders/m3/5574nnhn0yj488ccryqr7tc80000gn/T'
\end{DoxyCode}
