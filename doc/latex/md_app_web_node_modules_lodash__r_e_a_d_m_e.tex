\end{DoxyCode}

\begin{DoxyCode}

\end{DoxyCode}


In \href{http://nodejs.org/}{\tt Node.\+js} and \href{http://ringojs.org/}{\tt Ringo\+JS v0.\+8.\+0+}\+:


\begin{DoxyCode}

\end{DoxyCode}

\begin{DoxyCode}

\end{DoxyCode}


In \href{http://www.mozilla.org/rhino/}{\tt Rhino}\+:


\begin{DoxyCode}
