\begin{quote}
Left pad each line in a string \end{quote}




\subsection*{Install}


\begin{DoxyCode}
$ npm install --save lpad
\end{DoxyCode}


\subsection*{Example}


\begin{DoxyCode}
var lpad = require('lpad');

var str = 'foo\(\backslash\)nbar';
/*
foo
bar
*/

lpad(str, '    ');
/*
    foo
    bar
*/
\end{DoxyCode}


\subsection*{A\+PI}

\subsubsection*{lpad(string, pad)}

Pads each line in a string with the supplied pad string.
