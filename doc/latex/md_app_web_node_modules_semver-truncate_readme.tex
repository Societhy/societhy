\subsection*{Install}


\begin{DoxyCode}

\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
const semverTruncate = require('semver-truncate');

semverTruncate('1.2.3-foo', 'patch');
//=> '1.2.3'

semverTruncate('1.2.3', 'minor');
//=> '1.2.0'

semverTruncate('1.2.3', 'major');
//=> '1.0.0'
\end{DoxyCode}
