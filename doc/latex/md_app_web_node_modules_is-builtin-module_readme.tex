\begin{quote}
Check if a string matches the name of a Node.\+js builtin module \end{quote}


\subsection*{Install}


\begin{DoxyCode}
$ npm install --save is-builtin-module
\end{DoxyCode}


\subsection*{Usage}


\begin{DoxyCode}
var isBuiltinModule = require('is-builtin-module');

isBuiltinModule('fs');
//=> true

isBuiltinModule('unicorn');
//=> false :(
\end{DoxyCode}
